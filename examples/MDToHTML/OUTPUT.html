def factorial(n):
    if n <= 1:
        return 1
    return factorial(n-1) * n

Similarly, I can get Haskell syntax highlighting, if the parser supports it.

factorial :: Integer -> Integer
factorial n
    | n > 1 = n * factorial(n - 1)
    | otherwise = 1
